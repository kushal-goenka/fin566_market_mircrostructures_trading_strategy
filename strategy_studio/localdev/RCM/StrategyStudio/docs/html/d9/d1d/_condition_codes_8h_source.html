214 inline TradeCondition ConvertTVItchCrossingSessionTradeCodeToTradeCond(char crossing)
215 {
216  switch (crossing) {
217  case 'O':
218  return TRADE_COND_OB_OPENING_CROSSING_SESSION;
219  case 'C':
220  return TRADE_COND_OB_CLOSING_CROSSING_SESSION;
221  case 'H':
222  return TRADE_COND_OB_HALT_CROSSING_SESSION;
223  case 'I':
224  return TRADE_COND_OB_INTRADAY_CROSSING_SESSION;
225  default:
226  return TRADE_COND_UNKNOWN;
227  }
