370 inline bool IsValidPriceIncrement(const Instrument* instrument, double price)
371 {
372  double min_tick_size = instrument->min_tick_size(price);
373  return fabs(boost::math::round(price / min_tick_size) * min_tick_size - price) <= Utilities::PRICE_EPSILON * min_tick_size;
374 }

379 inline double RoundPriceToNearestTick(const Instrument* instrument, double price) {
380  double min_tick_size = instrument->min_tick_size(price);
381  double epsilon = price < 0 ? -min_tick_size / 2 : min_tick_size / 2;
382  StrategyStudioInt64 ticks = static_cast<StrategyStudioInt64>((price + epsilon) / min_tick_size);
383  return (ticks * min_tick_size);
384 }
