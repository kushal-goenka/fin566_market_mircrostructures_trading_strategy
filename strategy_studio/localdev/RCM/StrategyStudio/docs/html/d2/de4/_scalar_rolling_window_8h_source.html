138  void push_back(T val)
139  {
140  size_type N = c.size();
141  StatsType delta = 0;
142  if (c.full()) {
143  StatsType f=c.front();
144  --N;
145  m_rolling_sum -= f;
146  delta = (f - m_online_mean);
147  m_online_mean = (N > 0) ? m_online_mean - delta / N : 0;
148  m_online_m2 = (N > 0) ? m_online_m2 - delta * (f - m_online_mean) : 0;
149  }
150 
151  c.push_back(val);
152  ++N;
153  m_rolling_sum += val;
154  delta = (val - m_online_mean);
155  m_online_mean = (N < 2) ? val : m_online_mean + delta / N;
156  m_online_m2 = m_online_m2 + delta * (val - m_online_mean);
157  }
158 
164  void clear()
165  {
166  c.clear();
167  m_rolling_sum = 0;
168  m_online_mean = 0;
169  m_online_m2 = 0;
170  }
171 
177  StatsType Mean() const
178  {
179  return m_online_mean;//(m_rolling_sum / static_cast<StatsType>(c.size()));
180  }
181 
187  StatsType StdDev() const
188  {
189  size_type N = c.size();
190  return ((N > 1 && m_online_m2 > 0) ? std::sqrt(m_online_m2 / (N - 1)) : 0); //return the unbiased estimator
191  }
192 
196  StatsType Sum() const
197  {
198  return m_rolling_sum;
199  }
200 
207  StatsType ZScore() const
208  {
209  return (c.back() - Mean()) / StdDev();
210  }
211 
218  StatsType ZScore(size_type index) const
219  {
220  return (c[index] - Mean()) / StdDev();
221  }
222
