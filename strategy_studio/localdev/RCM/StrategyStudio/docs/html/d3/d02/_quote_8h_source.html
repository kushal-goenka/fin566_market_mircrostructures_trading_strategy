252 inline double Quote::micro_price() const
253 {
254  return (m_bidEntry.price() * m_askEntry.size() + m_askEntry.price() * m_bidEntry.size()) / static_cast<double>(m_bidEntry.size() + m_askEntry.size());
255 }
