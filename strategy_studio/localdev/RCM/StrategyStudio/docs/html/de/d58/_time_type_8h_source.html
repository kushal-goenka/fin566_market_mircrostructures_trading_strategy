242 inline TimeType ConvertUSEasternToUTC(DateType date, const std::string& time, boost::date_time::dst_flags dst = boost::date_time::calculate)
243 {
244  typedef boost::date_time::local_adjustor<boost::posix_time::ptime, -5, boost::posix_time::us_dst> us_eastern;
245  int hours = 0;
246  int minutes = 0;
247  size_t pos = time.find(':');
248  if (pos > 0) {
249  hours = atoi(time.substr(0, pos-1).c_str());
250  minutes = atoi(time.substr(pos).c_str());
251  }
252  else
253  hours = atoi(time.c_str());
254  return us_eastern::local_to_utc(TimeType(date, boost::posix_time::hours(hours) + boost::posix_time::minutes(minutes)), dst);
255 }
