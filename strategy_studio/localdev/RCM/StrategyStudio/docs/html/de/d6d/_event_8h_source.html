101 inline bool Event::OnEvent(EventMsg* eventMsg)
102 {
103  bool retval = false;
104  EventListenersIter end_it = m_eventListeners.end();
105  for(EventListenersIter it = m_eventListeners.begin(); it != end_it; ++it) {
106  eventMsg->PushActiveEvent(this);
107 
108  SERVER_DEBUG_LOG("Sending event to listener " + it->second->event_listener_name());
109  retval |= it->second->OnEvent(eventMsg);
110  SERVER_DEBUG_LOG("Finished sending event to listener " + it->second->event_listener_name());
111 
112  eventMsg->PopActiveEvent();
113  }
114 
115  return retval;
116 }
