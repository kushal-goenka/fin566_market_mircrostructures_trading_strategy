116 inline void CopyFilesInDirToTargetDir(const std::string& sourceDir,
117  const std::string& targetDir,
118  const std::string& fileExtension)
119 {
120  using namespace boost::filesystem;
121 
122  // precondition: fileExtension needs to start with .
123  assert(fileExtension.size() > 1 && fileExtension[0] == '.');
124  // precondition: targetDir needs to end with /
125  assert(targetDir[targetDir.size() - 1]);
126 
127  path sourcePath(sourceDir);
128 
129  // Create the directory if it does not exist
130  if(!boost::filesystem::exists(targetDir))
131  boost::filesystem::create_directory(targetDir);
132 
133  directory_iterator itr(sourcePath);
134  directory_iterator end_itr;
135 
136  // process all matching files within path
137  while (itr != end_itr) {
138  if (!is_directory(*itr)) {
139  path completePath = system_complete(*itr);
140 
141  // verify that this is a dll file or .so file (Linux)
142  std::string fileName = completePath.string();
143 
144  if (fileName.find(fileExtension) == std::string::npos) {
145  ++itr;
146  continue;
147  }
148 
149  std::string filename = completePath.filename().string();
150  std::string targetPath = targetDir + filename;
151 
152  copy_file(completePath, targetPath, copy_option::overwrite_if_exists);
153  }
154 
155  ++itr;
156  }
157 }

411 inline char* trim(char* buf, char delim = ' ') {
412  int len = static_cast<int>(strlen(buf));
413  if (len < 2) {
414  return buf;
415  }
416 
417  int i = 0;
418  int stripFromEnd = 0;
419  int stripFromBeg = 0;
420  //char* temp_buf = new char(len + 1);
421 
422  i = len - 1;
423  while( (buf[i] == delim) && (--i > 0) ) {
424  ++stripFromEnd;
425  }
426 
427  i = 0;
428  while( (buf[i] == delim) && (++i < len) ) {
429  ++stripFromBeg;
430  }
431 
432  int new_len = len - (stripFromBeg + stripFromEnd);
433  if (new_len < 1) {
434  buf[0] = '\0';
435  return buf;
436  }
437 
438  if ( (stripFromEnd > 0) || (stripFromBeg > 0) ) {
439  strncpy(buf, &buf[stripFromBeg], new_len);
440  buf[new_len] = '\0';
441  }
442 
443  return buf;
444 }

574  tokenizer tokens(text, sep);
