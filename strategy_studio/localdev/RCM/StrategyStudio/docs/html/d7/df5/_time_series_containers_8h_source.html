447  typename TimeWindow::iterator insert(const typename TimeWindow::value_type& val)
448  {
449  //delete values from front that fall out of range
450  while (c.size()>0 && cmp(c.front().first+m_window_size, val.first)) {

455  //if we're trying to inserting to the beginning but the resulting timespan is too big insert nothing and return iterator to begin()

457  return begin();
458  }
