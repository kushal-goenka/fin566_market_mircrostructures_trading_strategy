301 inline Utilities::TimeType NextAlignedTimeSecs(Utilities::TimeType currTime,
302  double alignSeconds,
303  Utilities::TimeType openTime)
304 {
305  // the minimum granularity we want to align to is per second
306  int alignTime = ((alignSeconds < 1) ? 1 : static_cast<int>(alignSeconds));
307 
308  boost::posix_time::time_duration td = currTime - openTime;
309  long secDiff = td.total_seconds();
310  int remainder = std::abs(secDiff) % alignTime;
311  long adjSeconds = (secDiff < 0) ? ((!remainder) ? alignTime : remainder) :
312  (alignTime - remainder);
313  boost::posix_time::time_duration adjTD = boost::posix_time::seconds(adjSeconds);
314  return currTime + adjTD;
315 }

317 inline Utilities::TimeType NextAlignedTimeMilliSecs(Utilities::TimeType currTime,
318  int alignMilliSeconds,
319  Utilities::TimeType openTime)
320 {
321  boost::posix_time::time_duration td = currTime - openTime;
322  StrategyStudioInt64 msecDiff = td.total_milliseconds();
323  StrategyStudioInt64 remainder = std::abs(msecDiff) % alignMilliSeconds;
324  StrategyStudioInt64 adjMilliSeconds = (msecDiff < 0) ? ((!remainder) ? alignMilliSeconds : remainder) :
325  (alignMilliSeconds - remainder);
326  boost::posix_time::time_duration adjTD = boost::posix_time::milliseconds(adjMilliSeconds);
327  return currTime + adjTD;
328 }
