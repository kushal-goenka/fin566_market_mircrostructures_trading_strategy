51 enum DepthReasonForChange {
52  DEPTH_REASON_FOR_CHANGE_UNKNOWN = 0,
53  DEPTH_REASON_FOR_CHANGE_UNATTRIBUTED_CHANGE = 1, // for use by adapters only, order book will attempt to classify Reason. DEPTH_UPDATE_TYPE_BY_ORDER modifications other than to reduce size lose queue position unless priority_indicator is input as a non-zero value and indicates otherwise
54  DEPTH_REASON_FOR_CHANGE_ADD_ORDER = 2, // new order added. if DEPTH_UPDATE_TYPE_BY_PRICE, also used to indicate a new price level
55  DEPTH_REASON_FOR_CHANGE_PARTIAL_CANCEL = 3, // modification to size as result of partial cancel or if executions not flagged by data source, remaining quantity maintains priority
56  DEPTH_REASON_FOR_CHANGE_FULL_CANCEL = 4, // deletion of order due to full cancellation or if executions not identified by data source. if DEPTH_UPDATE_TYPE_BY_PRICE, also used for deleted price levels if reason is unattributed by data source
57  DEPTH_REASON_FOR_CHANGE_EXECUTED = 5, // deletion or reduction of order due to execution, remaining quantity maintains priority
58  DEPTH_REASON_FOR_CHANGE_EXECUTED_AT_PRICE = 6, // deletion or reduction of order due to execution at diff price, remaining quantity maintains priority
59  DEPTH_REASON_FOR_CHANGE_MODIFY = 7, // cancel/replace without generating new order id, priority_indicator field specifies whether to reset queue priority if price unchanged. if DEPTH_UPDATE_TYPE_BY_PRICE, also used for modified price levels if reason is unattributed by data source
60  DEPTH_REASON_FOR_CHANGE_CANCEL_REPLACE = 8, // cancel/replace that generates new order id, order loses priority.
61 };
