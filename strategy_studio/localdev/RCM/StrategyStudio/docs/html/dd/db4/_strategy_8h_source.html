487 inline bool Strategy::OnExternalOrder(ExternalOrderEventMsg& msg)
488 {
489  bool result = false;
490 
491  if (msg.source() == ExternalOrderEventMsg::EXTERNAL_ORDER_SOURCE_STRATEGY_MANAGER) {
492  switch (msg.order_request_type()) {
493  case FlashProtocol::ORDER_SUBMIT_NEW: {
494  TradeActionResult taResult = trade_actions()->SendNewOrder(msg.params(), msg.account().empty() ? account_owner() : msg.account());
495  if (taResult == TRADE_ACTION_RESULT_SUCCESSFUL) {
496  result = true;
497  } else {
498  std::string message = "Client NEW Order with ID = " + msg.client_order_id() + ", Rejected with status = " + DataTypesUtils::TradeActionResultToString(taResult);
499  logger().LogToClient(Utilities::LOGLEVEL_ERROR, message);
500  }
501  break;
502  }
503  case FlashProtocol::ORDER_CANCEL: {
504  TradeActionResult taResult = trade_actions()->SendCancelOrder(msg.params().order_id);
505  if (taResult == TRADE_ACTION_RESULT_SUCCESSFUL) {
506  result = true;
507  } else {
508  std::string message = "Client CANCEL Order with ID = " + msg.client_order_id() + ", Rejected with status = " + DataTypesUtils::TradeActionResultToString(taResult);
509  logger().LogToClient(Utilities::LOGLEVEL_ERROR, message);
510  }
511  break;
512  }
513  case FlashProtocol::ORDER_REPLACE: {
514  TradeActionResult taResult = trade_actions()->SendCancelReplaceOrder(msg.params().order_id, msg.params());
515  if (taResult == TRADE_ACTION_RESULT_SUCCESSFUL) {
516  result = true;
517  } else {
518  std::string message = "Client CANCEL REPLACE Order with ID = " + msg.client_order_id() + ", Rejected with status = " + DataTypesUtils::TradeActionResultToString(taResult);
519  logger().LogToClient(Utilities::LOGLEVEL_ERROR, message);
520  }
521  break;
522  }
523  default:
524  break;
525  }
526  }
527  return result;
528 }
